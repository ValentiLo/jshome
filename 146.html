140. Пример: Создать два объекта-коллекции (car и moto) со свойствами color и brand (у объекта moto еще свойство horse power). Вывести значение свойства color у объекта car и свойства brand у объекта moto.
 Решение:
let car = new Object();
car.color = "Белый";
car.maxSpeed=220;
car.brand = "Nissan";
document.write("У машины цвет: " + car.color);
let moto = {
        color: "Синий",
        "horse power": 130, // если свойство из двух слов - используем кавычки
        brand: "Yamaha"
        };
 document.write("У мотоцикла производитель: " + moto.brand);
